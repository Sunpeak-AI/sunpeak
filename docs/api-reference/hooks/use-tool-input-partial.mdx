## Overview

Receives streaming partial input arguments as the AI model generates them. Useful for showing live previews of what the model is about to send.

## Import

```tsx
import { useToolInputPartial } from 'sunpeak';
```

## Signature

```tsx
function useToolInputPartial(
  app: App | null
): ToolInputPartial | null
```

## Parameters

<ResponseField name="app" type="App | null" required>
  App instance from `useApp()`.
</ResponseField>

## Returns

<ResponseField name="return" type="ToolInputPartial | null">
  Partial tool input or null if not streaming.
</ResponseField>

### ToolInputPartial

<ResponseField name="arguments" type="Record<string, unknown>">
  Partial arguments being streamed by the AI model.
</ResponseField>

## Usage

```tsx
import { useApp, useToolInputPartial } from 'sunpeak';

function MyResource() {
  const { app } = useApp({ appInfo: { name: 'MyResource', version: '1.0.0' } });
  const partial = useToolInputPartial(app);

  if (partial?.arguments) {
    return <div>Preview: {JSON.stringify(partial.arguments)}</div>;
  }

  return <div>Waiting for input...</div>;
}
```

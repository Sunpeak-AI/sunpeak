## Overview

The `useDisplayMode` hook returns the current display mode of the app. It is a convenience wrapper around `useHostContext`.

## Import

```tsx
import { useDisplayMode } from 'sunpeak';
```

## Signature

```tsx
function useDisplayMode(app: App | null): McpUiDisplayMode
```

## Parameters

<ResponseField name="app" type="App | null" required>
  The App instance from `useApp()`.
</ResponseField>

## Returns

<ResponseField name="return" type="'inline' | 'pip' | 'fullscreen'">
  The current display mode. Defaults to `'inline'` when unavailable.
</ResponseField>

## Usage

```tsx
import { useApp, useDisplayMode } from 'sunpeak';

function MyResource() {
  const { app } = useApp({ appInfo: { name: 'MyResource', version: '1.0.0' } });
  const displayMode = useDisplayMode(app);

  return (
    <div className={displayMode === 'fullscreen' ? 'h-screen' : 'max-h-96'}>
      Current mode: {displayMode}
    </div>
  );
}
```

description: "Get the current color theme."
---

## Overview

The `useTheme` hook returns the host's current theme. It is a convenience wrapper around `useHostContext`.

## Import

```tsx
import { useTheme } from 'sunpeak';
```

## Signature

```tsx
function useTheme(app: App | null): McpUiTheme
```

## Parameters

<ResponseField name="app" type="App | null" required>
  The App instance from `useApp()`.
</ResponseField>

## Returns

<ResponseField name="return" type="'light' | 'dark'">
  The current theme. Defaults to `'light'` when unavailable.
</ResponseField>

## Usage

```tsx
import { useApp, useTheme } from 'sunpeak';

function MyResource() {
  const { app } = useApp({ appInfo: { name: 'MyResource', version: '1.0.0' } });
  const theme = useTheme(app);

  return (
    <div className={theme === 'dark' ? 'bg-black text-white' : 'bg-white text-black'}>
      ...
    </div>
  );
}
```
